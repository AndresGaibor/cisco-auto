SCENARIO: Configure interface with paging

BEFORE:
  term.enterCommand('interface Gi0/0')
  for i in range(30):
    if '--More--' in output:
      term.enterCommand(' ')
    if '#' in output and len > 20:
      break  # ← Hope this is the real prompt!
  
  if '% Invalid' in output:
    return { ok: false }
  else:
    return { ok: true }  # ← Could be false positive!

AFTER:
  engine = IosSessionEngine()
  engine.processEvent({ type: 'commandStarted', command: 'interface Gi0/0' })
  
  # State machine automatically handles:
  # - Detects paging prompt → paging state
  # - Sends space → awaiting-output state
  # - Detects mode change → updates mode
  # - Detects command end → completed state
  
  result = {
    ok: engine.isComplete(),
    diagnostics: {
      source: 'terminal',
      completionReason: 'command-ended',
      errors: []
    },
    session: { mode: 'config' },  # ← Know exact mode!
    interaction: { pagesAdvanced: 1, modesChanged: 1 }  # ← Track what happened!
  }
